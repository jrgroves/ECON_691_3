working <- ufo %>%
  select(datetime, city, state, country, latitude, longitude) %>%
  filter(country == "us") %>%
  mutate(year = str_split_i(str_split_i(datetime, " ", 1), "/", 3)) %>%
  filter(state == "IN" | state == "IL" | state == "KY",
         year == 2010) %>%
  count(state)

census <- acs %>%
  select(!c(NAME, moe)) %>%
  mutate(variable = case_when(variable == "B01002_001" ~ "median.age",
                              variable == "B01003_001" ~ "population",
                              TRUE ~ variable),
         state = case_when(GEOID == 17 ~ "IL",
                           GEOID == 18 ~ "IN",
                           TRUE ~ "KY")) %>%
  pivot_wider(id_cols = c(GEOID, state, geometry), 
              names_from = variable, 
              values_from = estimate) %>%
  mutate(population = population / 100000)

core <- census %>%
  left_join(., working, by = "state")%>%
  mutate(sight_per = n / population)%>%
  relocate(geometry, .after = sight_per)

Visualization

  • Data visualization is just as important as the downloading and cleaning of data because it allows your reader to quickly get a feel for your data.
  • Visualization also serves as a way for you to ensure that your data is “clean” and there are not potential errors that you have missed that might skew your results
  • Three key ways to visualize
    • Graphs
    • Plots (Maps)
    • Tables

Visualization: Graphs

  • There are several ways to create visualizations in R, but we will use the ggplot2 package that is part of the tidyverse package.
  • ggplot2 is the standard at this point in time for visualization given the extreme customization ability.
  • One of the first visualizations we will look at is the Box-Whisker Plot, or the boxplot.
  • ggplot is based on a series of geoms that allow you to produce a variety of different base graphics.
  • Common geoms include: line, point, boxplot, and sf.

GGPLOT Coding

ggplot(core) + 
  geom_boxplot(aes(x = year, y = sightings), color = "navy") +
  theme_bw()
  1. This line is calling the general ggplot() command and is a place where you can set any variables that will be constant throughout the code. In this case, we are setting the dataframe that we will be using.
  2. This line is calling the specific graph we want on THIS LAYER. Next is the aes() which is the option for aesthetics or “looks” of the graph. We set our x-axis and y-axis variables and we can also denotes groups if we want here. Notice that outside the aes() command we change the color of the graph. This is okay here because we want it for EVERYTHING in the plot.
    • ggplot creates layers and the first listed is the top layer. This is important to know when we are overlaying different graphs. You should treat each geom_ line a separate layer.
  3. Themes are how ggplot sets many of the visual elements. In this case we are using a pre-set theme called bw, but we can run the code without to see the difference.

Regressions - Linear Model

  • When we view the code below, we see the action takes place in line that call a second command, lm.fit
function (formula, data, subset, weights, na.action, method = "qr", 
    model = TRUE, x = FALSE, y = FALSE, qr = TRUE, singular.ok = TRUE, 
    contrasts = NULL, offset, ...) 
{
    ret.x <- x
    ret.y <- y
    cl <- match.call()
    mf <- match.call(expand.dots = FALSE)
    m <- match(c("formula", "data", "subset", "weights", "na.action", 
        "offset"), names(mf), 0L)
    mf <- mf[c(1L, m)]
    mf$drop.unused.levels <- TRUE
    mf[[1L]] <- quote(stats::model.frame)
    mf <- eval(mf, parent.frame())
    if (method == "model.frame") 
        return(mf)
    else if (method != "qr") 
        warning(gettextf("method = '%s' is not supported. Using 'qr'", 
            method), domain = NA)
    mt <- attr(mf, "terms")
    y <- model.response(mf, "numeric")
    w <- as.vector(model.weights(mf))
    if (!is.null(w) && !is.numeric(w)) 
        stop("'weights' must be a numeric vector")
    offset <- model.offset(mf)
    mlm <- is.matrix(y)
    ny <- if (mlm) 
        nrow(y)
    else length(y)
    if (!is.null(offset)) {
        if (!mlm) 
            offset <- as.vector(offset)
        if (NROW(offset) != ny) 
            stop(gettextf("number of offsets is %d, should equal %d (number of observations)", 
                NROW(offset), ny), domain = NA)
    }
    if (is.empty.model(mt)) {
        x <- NULL
        z <- list(coefficients = if (mlm) matrix(NA_real_, 0, 
            ncol(y)) else numeric(), residuals = y, fitted.values = 0 * 
            y, weights = w, rank = 0L, df.residual = if (!is.null(w)) sum(w != 
            0) else ny)
        if (!is.null(offset)) {
            z$fitted.values <- offset
            z$residuals <- y - offset
        }
    }
    else {
        x <- model.matrix(mt, mf, contrasts)
        z <- if (is.null(w)) 
            lm.fit(x, y, offset = offset, singular.ok = singular.ok, 
                ...)
        else lm.wfit(x, y, w, offset = offset, singular.ok = singular.ok, 
            ...)
    }
    class(z) <- c(if (mlm) "mlm", "lm")
    z$na.action <- attr(mf, "na.action")
    z$offset <- offset
    z$contrasts <- attr(x, "contrasts")
    z$xlevels <- .getXlevels(mt, mf)
    z$call <- cl
    z$terms <- mt
    if (model) 
        z$model <- mf
    if (ret.x) 
        z$x <- x
    if (ret.y) 
        z$y <- y
    if (!qr) 
        z$qr <- NULL
    z
}

Regressions - Linear Model

  • Again we notice that this is not the actual estimation of the model, but another shell command that ensure the data and method is legitimate and then we see the command, .Call
  • This is a way to call a C-Function from R and here the function called is a C based version of dqrls which stands for “QR decomposition for least squares”.
  • From here we can dig further, but we at least know how this method is working.
function (x, y, offset = NULL, method = "qr", tol = 1e-07, singular.ok = TRUE, 
    ...) 
{
    if (is.null(n <- nrow(x))) 
        stop("'x' must be a matrix")
    if (n == 0L) 
        stop("0 (non-NA) cases")
    p <- ncol(x)
    if (p == 0L) {
        return(list(coefficients = numeric(), residuals = y, 
            fitted.values = 0 * y, rank = 0, df.residual = length(y)))
    }
    ny <- NCOL(y)
    if (is.matrix(y) && ny == 1) 
        y <- drop(y)
    if (!is.null(offset)) 
        y <- y - offset
    if (NROW(y) != n) 
        stop("incompatible dimensions")
    if (method != "qr") 
        warning(gettextf("method = '%s' is not supported. Using 'qr'", 
            method), domain = NA)
    chkDots(...)
    z <- .Call(C_Cdqrls, x, y, tol, FALSE)
    if (!singular.ok && z$rank < p) 
        stop("singular fit encountered")
    coef <- z$coefficients
    pivot <- z$pivot
    r1 <- seq_len(z$rank)
    dn <- colnames(x) %||% paste0("x", 1L:p)
    nmeffects <- c(dn[pivot[r1]], rep.int("", n - z$rank))
    r2 <- if (z$rank < p) 
        (z$rank + 1L):p
    else integer()
    if (is.matrix(y)) {
        coef[r2, ] <- NA
        if (z$pivoted) 
            coef[pivot, ] <- coef
        dimnames(coef) <- list(dn, colnames(y))
        dimnames(z$effects) <- list(nmeffects, colnames(y))
    }
    else {
        coef[r2] <- NA
        if (z$pivoted) 
            coef[pivot] <- coef
        names(coef) <- dn
        names(z$effects) <- nmeffects
    }
    z$coefficients <- coef
    r1 <- y - z$residuals
    if (!is.null(offset)) 
        r1 <- r1 + offset
    if (z$pivoted) 
        colnames(z$qr) <- colnames(x)[z$pivot]
    qr <- z[c("qr", "qraux", "pivot", "tol", "rank")]
    c(z[c("coefficients", "residuals", "effects", "rank")], list(fitted.values = r1, 
        assign = attr(x, "assign"), qr = structure(qr, class = "qr"), 
        df.residual = n - z$rank))
}

Regressions - Non-Package Methods

  • If we wanted to regression a model using the maximum likelihood approach, we can do that in the script as well.
    • We would define our likelihood function (or negative likelihood function)
    • We would use the command optimize()
  • If we wanted to do a simple linear regression using matrix math, we could also do that in the script.
    • Translate our numerical data frame into a matrix using as.Matrix()
    • Perform the required matrix math for linear regressions.
  • No matter how you or the package choose to restimate the model, it is imperative that you do not just download something, plug data in, and run it. You will generally always get an answer, but it may not be the “right” one.
